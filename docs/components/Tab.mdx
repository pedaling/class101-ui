## onTabChange

<Playground>
  {() => {
    const value = 'PREVIEW';
    const tabs = [
      {
        value: 'PREVIEW',
        title: '알림 신청',
      },
      {
        value: 'KLASS',
        title: '클래스 및 패키지',
      },
      {
        value: 'CURRICULM',
        title: '커리큘럼',
      },
    ];
    class Example extends React.Component {
      constructor(props) {
        super(props);
        this.state = { currentValue: 'PREVIEW' };
      }
      render() {
        let SectionElement;
        switch (this.state.currentValue) {
          case 'PREVIEW':
            SectionElement = <div>알림 신청 알림 신청 알림 신청</div>
            break;
          case 'KLASS':
            SectionElement = <div>클래스 및 패키지 클래스 및 패키지 클래스 및 패키지</div>
            break;
          default:
            SectionElement = <div>커리큘럼 커리큘럼 커리큘럼</div>
            break;
        }
        return (
          <Tab onTabChange={(value) => {this.setState({currentValue: value})}} tabs={tabs} currentValue={this.state.currentValue}>
            {SectionElement}
          </Tab>
        );
      }
    }
    return <Example />

}}

</Playground>
