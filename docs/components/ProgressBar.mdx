# Progress Bar

진행 상황의 정도를 보여주기 유용한 컴포넌트입니다.

## Props

<PropsTable of={ProgressBar} />

## Basic usage

<Playground>
  <ProgressBar value={30} />
</Playground>

## Animation

<Playground>
  {() => {
    const [value, setValue] = useState(10);

    const handleClickToIncrease = useCallback(() => {
      if(value < 100) {
        setValue(value => value + 10);
      }
    }, [value])

    const handleClickToDecrease = useCallback(() => {
      if(value > 0) {
        setValue(value => value - 10);
      }
    }, [value])

    return (
      <>
        <ProgressBar value={value} />
        <br/>
        <ButtonGroup>
          <Button onClick={handleClickToIncrease}>+ 10%</Button>
          <Button onClick={handleClickToDecrease}>- 10%</Button>
        </ButtonGroup>
      </>
    )
  }}
</Playground>
