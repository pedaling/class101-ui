import { Section, GridList, Body1 } from '@class101/ui';
import { Playground, Props } from 'docz';
import { imageSrc } from './interface';
import { PlayGroundBanner } from '../docsComponents';

# Section

titleStyle은 Typography, buttonStyle은 Button 컴포넌트의 속성을 참고해주세요.

## Props

<Props of={Section} />

## Basic usage

섹션은 헤더를 반드시 가지고 있어야 합니다.

<Playground>
  <Section title="가장 완강률이 높은 클래스">
    <PlayGroundBanner>리스트나 캐러셀 등이 들어갈 수 있습니다.</PlayGroundBanner>
  </Section>
</Playground>

## Anchor or Action to Title

`to`, `onClick`을 지정할 경우 섹션 헤더는 액션이 가능한 요소가 됩니다.  
BreakPoint에 따라 'lg'에서는 오버시에 나오고 'sm'에서는 항상 보이는 요소입니다.

<Playground>
  <Section title="가장 완강률이 높은 클래스" to="/">
    <GridList
      items={[
        {
          id: 1,
          title: '리스트 아이템 1',
          image: imageSrc,
          to: '/',
        },
        {
          id: 2,
          title: '리스트 아이템 2',
          image: imageSrc,
          to: '/',
        },
      ]}
      smColumn={2}
      renderItem={item => {
        const { title, image } = item;
        return (
          <div>
            <div>
              <img style={{ width: '100%' }} src={image} alt={title} />
            </div>
            <Body1>{title}</Body1>
          </div>
        );
      }}
    />
  </Section>
</Playground>

## with LinkText

BreakPoint에 따라 'sm'에서는 사라집니다.
