## Basic usage

<Playground>
  {() => {
    const [opened, setOpened] = React.useState(false);
    const openComboBox = () => setOpened(true);
    const closeComboBox = () => setOpened(false);
    return(
      <>
        <ComboBox 
          items={[
            { label: '수정하기', icon: <Icon.EditOutline /> },
            { label: '알림 끄기', description: '해당 게시물의 메일과 푸시알림을 받지 않습니다.', icon: <Icon.BellOff /> },
            { label: '삭제하기', textColor: Colors.red500 },
          ]}
          opened={opened}
          onOpen={openComboBox}
          onClose={closeComboBox}
        >
          <IconButton icon={<Icon.MoreHorizontal />} />
        </ComboBox>
      </>
    )}}
</Playground>
