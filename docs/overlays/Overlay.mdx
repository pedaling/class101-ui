## Basic usage

<Playground>

    () => {
      function Example() {
        const [opened, setOpened] = React.useState(false);

        const openOverlay = () => setOpened(true);
        const closeOverlay = () => setOpened(false);

        return (
          <>
            <Button onClick={openOverlay}>Open Overlay</Button>
            <Overlay opened={opened} onClose={closeOverlay}>
              Overlay Children
            </Overlay>
          </>
        )
      }
      return <Example />
    }
